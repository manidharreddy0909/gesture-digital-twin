### 2A. CAMERA CALIBRATION

```python
from camera_calibration import CameraCalibrator, CoordinateTransformer

# Quick calibration from screen size
calibration = CameraCalibrator.estimate_from_image_size(
    width=1920, height=1080, fov_degrees=60.0
)

# Transform hand landmarks to 3D world

world_coords = transformer.transform_landmarks(
    normalized_landmarks  # From MediaPipe
)
```

### 2B. 3D CURSOR FILTERING

```python
from cursor_controller_3d import CursorController3D

# Create 3D controller with Kalman filter
controller = CursorController3D(
    use_enhanced_kalman=True,  # 9D acceleration model
    adaptive_smoothing=True,
    kalman_adaptive_noise=True
)

# Update cursor (replacing 2D version)
filtered_pos = controller.update_cursor_3d(
    hand_id=0,
    x_norm=0.5, y_norm=0.5, z_norm=0.5,
    timestamp=time.time()
)
```

### 2C. ROBOT ARM CONTROL

```python
from robot_controller import RobotArmUR5, HandToArmMapper

# Initialize robot
robot = RobotArmUR5("192.168.1.100")  # IP of UR5
robot.connect()

# Map hand to arm
mapper = HandToArmMapper(position_scale=1.0, max_velocity=0.5)
target = mapper.map_hand_to_arm_target(left_hand, right_hand, gestures)

# Move!
robot.move_to_position(target)
```

### 2D. 3D OBJECT MANIPULATION

```python
from object_manipulator import ObjectManipulator, Object3D

# Create scene
manipulator = ObjectManipulator()
cube = Object3D(id="cube", position=np.array([0.5, 0.5, 0.5]))
manipulator.add_object(cube)

# Interact with gestures
manipulator.select_object_at(hand_position)
manipulator.apply_gesture("pinch", hand_position)
manipulator.apply_gesture("zoom_in", two_hand_gesture="zoom_in")

# Simulate physics
manipulator.update_physics(dt=0.033, damping=0.95)
```

### 2E. UNREAL ENGINE INTEGRATION

```python
from unreal_bridge import UnrealPythonAPIBridge, HandSkeletonConverter

```
